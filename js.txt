var hw = "Hello world!"

// Tamanho de uma string/(var)
"Hello world!".length                                                   // 12
hw.length                                                               // 12

// Ver um carater específico de uma string/(var), através do index
"Hello world!"[0]                                                       // H
("hello" + "world!")[6]                                                 // o (pois começa a contar a partir do 0)
hw[0]                                                                   // H

// Ver o último carater de uma string/(var)
hw[hw.length - 1]                                                       // !
